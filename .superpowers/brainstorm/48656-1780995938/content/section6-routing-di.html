Auth + role guard

route_guard.dart
String? redirect(BuildContext ctx, GoRouterState state) { final auth = ctx.read<AuthBloc>().state; final isPublic = _publicRoutes .contains(state.matchedLocation); if (auth is AuthSuccess) { if (isPublic) { // already logged in → send home return auth.user.role == Role.vendor ? '/vendor/home' : '/buyer/home'; } // block buyer from /vendor/* and vice versa if (state.matchedLocation.startsWith('/vendor') && auth.user.role != Role.vendor) return '/buyer/home'; return null; // allow } return isPublic ? null : '/login'; }
